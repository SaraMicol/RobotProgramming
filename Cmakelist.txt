cmake_minimum_required(VERSION 3.0.2)
project(ros_2d_multi_robot_simulator)

add_compile_options(-std=c++11)

find_package(catkin REQUIRED COMPONENTS
  roscpp
  geometry_msgs
  nav_msgs
  sensor_msgs
  tf2_ros
)

find_package(OpenCV REQUIRED)
find_package(yaml-cpp REQUIRED)

catkin_package()

include_directories(
  ${catkin_INCLUDE_DIRS}
  ${OpenCV_INCLUDE_DIRS}
  ${YAML_CPP_INCLUDE_DIRS}
)

add_executable(robsim_node
  src/robsim_node.cpp
  src/roblibs/grid_map.cpp
  src/roblibs/world_item.cpp
  src/roblibs/draw_helpers.cpp
  src/roblibs/laser_scanner.cpp
  src/roblibs/laser_scan.cpp
)

target_link_libraries(robsim_node
  ${catkin_LIBRARIES}
  ${OpenCV_LIBS}
  ${YAML_CPP_LIBRARIES}
)
